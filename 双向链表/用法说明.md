# 功能
* 往左端添加元素
* 往右端添加元素
* 删除一个链表中某值的元素
# 用法实例
```
#include<iostream>
#include"DbList.h"
using namespace std;
int main()
{
	cout << "测试双向链表：" << endl;
	DbList<int> ls;//建立空的双向链表
	//从左插入两个，再从右插两个，最后顺序为4，5，6，7
	ls.insert_left(5);
	ls.insert_left(4);
	ls.insert_right(6);
	ls.insert_right(7);
	MyIterator<int> iter(ls);//建立迭代器，用链表ls初始化
	cout << *iter.first() << endl;//迭代器指向第一个数据4
	//从左往右依次打印链表中的数据
	while (iter.right_not_null())
		cout << *iter.right() << endl;
	//从右往左依次打印链表中的数据
	while (iter.left_not_null())
		cout << *iter.left() << endl;
	return 0;
}
```
