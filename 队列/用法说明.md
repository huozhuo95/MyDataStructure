# 功能
* 检查队列是否为空
* 返回队首元素的值
* 返回队尾元素的值
* 添加一个元素放在队尾
* 从队首删除一个元素
# 用法实例
```
#include<iostream>
#include"MyQueue.h"
using namespace std;

int main()
{
	//创建队列的对象，容量为5(如果不设，默认为10)
	MyQueue<char> q1(5);
	cout << q1.is_empty() << endl; //打印1
	char a = 'a', b = 'b', c = 'c',d='d',e='e',f='f';
	//依次添加abc三个元素
	q1.push(a);
	q1.push(b);
	q1.push(c);
	//看是否为空队列，看队首元素，队尾元素
	cout << q1.is_empty() << endl; //打印0
	cout << q1.front_item() << endl; //打印a
	cout << q1.rear_item() << endl; //打印c
	//删除一个元素，再新增一个e
	q1.pop();
	q1.push(e);
	cout << endl;
	//看队首元素，队尾元素
	cout << q1.front_item() << endl; //打印b
	cout << q1.rear_item() << endl; //打印e
	//删除一个元素，再新增一个f
	q1.pop();
	q1.push(f);
	cout << endl;
	//看队首元素，队尾元素
	cout << q1.front_item() << endl; //打印c
	cout << q1.rear_item() << endl; //打印f
	//狂加元素，让它自动扩容
	q1.push('g');
	q1.push('h');
	q1.push('i');
	q1.push('j');
	q1.push('k');
	cout << endl;
	//看队首元素，队尾元素
	cout << q1.front_item() << endl; //打印c
	cout << q1.rear_item() << endl; //打印k
	return 0;
}
```
